server {
    server_name dashboard.jorapp.org;
    root /var/www/dashboard;
    index index.html;

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/dashboard.jorapp.org/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/dashboard.jorapp.org/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

}
server {
    if ($host = dashboard.jorapp.org) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    listen 80;
    server_name dashboard.jorapp.org;
    return 404; # managed by Certbot


}
