server {
    server_name db.jorapp.org;

    # Uploads de photos (vitrine visuelle)
    client_max_body_size 50M;

    location / {
        proxy_pass http://127.0.0.1:8091;
        proxy_http_version 1.1;

        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # Temps réel PocketBase (SSE) : pas de buffering, connexion longue
        proxy_set_header Connection '';
        proxy_buffering off;
        proxy_cache off;
        proxy_read_timeout 86400s;
        chunked_transfer_encoding off;
    }

    listen [::]:443 ssl; # managed by Certbot
    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/db.jorapp.org/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/db.jorapp.org/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

}
server {
    if ($host = db.jorapp.org) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    listen 80;
    listen [::]:80;
    server_name db.jorapp.org;
    return 404; # managed by Certbot


}
